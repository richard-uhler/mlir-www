38 // Extract an LLVM IR type from the LLVM IR dialect type.
39 static LLVM::LLVMType unwrap(Type type) {
40  if (!type)
41  return nullptr;
42  auto *mlirContext = type.getContext();
43  auto wrappedLLVMType = type.dyn_cast<LLVM::LLVMType>();
44  if (!wrappedLLVMType)
45  emitError(UnknownLoc::get(mlirContext),
46  "conversion resulted in a non-LLVM type");
47  return wrappedLLVMType;
48 }

50 static Optional<Type>
51 convertScalableVectorTypeToLLVM(ScalableVectorType svType,
52  LLVMTypeConverter &converter) {
53  auto elementType = unwrap(converter.convertType(svType.getElementType()));
54  if (!elementType)
55  return {};
56 
57  auto sVectorType =
58  LLVM::LLVMScalableVectorType::get(elementType, svType.getShape().back());
59  return sVectorType;
60 }
