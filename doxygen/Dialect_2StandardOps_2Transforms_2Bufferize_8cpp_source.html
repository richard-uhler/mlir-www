39 class BufferizeDynamicTensorFromElementsOp
40  : public OpConversionPattern<DynamicTensorFromElementsOp> {
41 public:
42  using OpConversionPattern::OpConversionPattern;
43 
44  LogicalResult
45  matchAndRewrite(DynamicTensorFromElementsOp op, ArrayRef<Value> operands,
46  ConversionPatternRewriter &rewriter) const final {
47  // Allocate memory.
48  Location loc = op.getLoc();
49  DynamicTensorFromElementsOp::Adaptor transformed(operands);
50  RankedTensorType tensorType = op.getType().cast<RankedTensorType>();
51  MemRefType memrefType =
52  MemRefType::get(tensorType.getShape(), tensorType.getElementType());
53  Value result =
54  rewriter.create<AllocOp>(loc, memrefType, transformed.dynamicExtents());
55 
56  // Collect loop bounds.
57  int64_t rank = tensorType.getRank();
58  Value zero = rewriter.create<ConstantIndexOp>(loc, 0);
59  Value one = rewriter.create<ConstantIndexOp>(loc, 1);
60  SmallVector<Value, 4> lowerBounds(rank, zero);
61  SmallVector<Value, 4> steps(rank, one);
62  SmallVector<Value, 4> upperBounds;
63  int nextDynamicIndex = 0;
64  for (int i = 0; i < rank; i++) {
65  Value upperBound =
66  tensorType.isDynamicDim(i)
67  ? transformed.dynamicExtents()[nextDynamicIndex++]
68  : rewriter.create<ConstantIndexOp>(loc, memrefType.getDimSize(i));
69  upperBounds.push_back(upperBound);
70  }
71 
72  // Generate tensor elements with a parallel loop that stores into
73  // each element of the resulting memref.
74  //
75  // This is a bit tricky. We cannot simply clone the ops because when an op
76  // is cloned, it must be legalized. However, we want to allow arbitrary ops
77  // in the body that we don't necessarily have legalization patterns for as
78  // part of this dialect conversion invocation.
79  //
80  // To accomplish this, we use mergeBlockBefore to "move" this op's body
81  // into the scf.parallel's body.
82  auto parallel =
83  rewriter.create<scf::ParallelOp>(loc, lowerBounds, upperBounds, steps);
84  Block *parallelBody = parallel.getBody();
85  rewriter.mergeBlockBefore(op.getBody(), parallelBody->getTerminator(),
86  parallelBody->getArguments());
87  // Replace the inlined yield op with a store op. The scf.parallel's builder
88  // already populated an scf.yield at the end, so we don't need to worry
89  // about creating that.
90  Operation *elementYield = parallelBody->getTerminator()->getPrevNode();
91  rewriter.setInsertionPointAfter(elementYield);
92  rewriter.replaceOpWithNewOp<StoreOp>(elementYield,
93  elementYield->getOperands()[0], result,
94  parallelBody->getArguments());
95 
96  rewriter.replaceOp(op, {result});
97  return success();
98  }
99 };
100 } // namespace

135 class BufferizeTensorFromElementsOp
136  : public OpConversionPattern<TensorFromElementsOp> {
137 public:
138  using OpConversionPattern::OpConversionPattern;
139  LogicalResult
140  matchAndRewrite(TensorFromElementsOp op, ArrayRef<Value> operands,
141  ConversionPatternRewriter &rewriter) const override {
142  int numberOfElements = op.elements().size();
143  auto resultType = MemRefType::get(
144  {numberOfElements}, op.getType().cast<TensorType>().getElementType());
145  Value result = rewriter.create<AllocOp>(op.getLoc(), resultType);
146  for (auto element : llvm::enumerate(op.elements())) {
147  Value index =
148  rewriter.create<ConstantIndexOp>(op.getLoc(), element.index());
149  rewriter.create<StoreOp>(op.getLoc(), element.value(), result, index);
150  }
151  rewriter.replaceOp(op, {result});
152  return success();
153  }
154 };

172 struct StdBufferizePass : public StdBufferizeBase<StdBufferizePass> {
173  void runOnFunction() override {
174  auto *context = &getContext();
175  BufferizeTypeConverter typeConverter;
176  OwningRewritePatternList patterns;
177  ConversionTarget target(*context);
178 
179  target.addLegalDialect<StandardOpsDialect>();
180  target.addLegalDialect<scf::SCFDialect>();
181 
182  populateStdBufferizePatterns(context, typeConverter, patterns);
183  target.addIllegalOp<DynamicTensorFromElementsOp, TensorCastOp,
184  TensorFromElementsOp>();
185  // We only bufferize the case of tensor selected type and scalar condition,
186  // as that boils down to a select over memref descriptors (don't need to
187  // touch the data).
188  target.addDynamicallyLegalOp<SelectOp>([&](SelectOp op) {
189  return typeConverter.isLegal(op.getType()) ||
190  !op.condition().getType().isa<IntegerType>();
191  });
192  target.addDynamicallyLegalOp<DimOp>(
193  [&](DimOp op) { return typeConverter.isLegal(op); });
194  if (failed(
195  applyPartialConversion(getFunction(), target, std::move(patterns))))
196  signalPassFailure();
197  }
198 };
