300 /// Fold constant dimensions into an alloc like operation.
301 template <typename AllocLikeOp>
302 struct SimplifyAllocConst : public OpRewritePattern<AllocLikeOp> {
303  using OpRewritePattern<AllocLikeOp>::OpRewritePattern;
304 
305  LogicalResult matchAndRewrite(AllocLikeOp alloc,
306  PatternRewriter &rewriter) const override {
307  // Check to see if any dimensions operands are constants. If so, we can
308  // substitute and drop them.
309  if (llvm::none_of(alloc.getOperands(), [](Value operand) {
310  return matchPattern(operand, m_ConstantIndex());
311  }))
312  return failure();
313 
314  auto memrefType = alloc.getType();
315 
316  // Ok, we have one or more constant operands. Collect the non-constant ones
317  // and keep track of the resultant memref type to build.
318  SmallVector<int64_t, 4> newShapeConstants;
319  newShapeConstants.reserve(memrefType.getRank());
320  SmallVector<Value, 4> newOperands;
321 
322  unsigned dynamicDimPos = 0;
323  for (unsigned dim = 0, e = memrefType.getRank(); dim < e; ++dim) {
324  int64_t dimSize = memrefType.getDimSize(dim);
325  // If this is already static dimension, keep it.
326  if (dimSize != -1) {
327  newShapeConstants.push_back(dimSize);
328  continue;
329  }
330  auto *defOp = alloc.getOperand(dynamicDimPos).getDefiningOp();
331  if (auto constantIndexOp = dyn_cast_or_null<ConstantIndexOp>(defOp)) {
332  // Dynamic shape dimension will be folded.
333  newShapeConstants.push_back(constantIndexOp.getValue());
334  } else {
335  // Dynamic shape dimension not folded; copy operand from old memref.
336  newShapeConstants.push_back(-1);
337  newOperands.push_back(alloc.getOperand(dynamicDimPos));
338  }
339  dynamicDimPos++;
340  }
341 
342  // Create new memref type (which will have fewer dynamic dimensions).
343  MemRefType newMemRefType =
344  MemRefType::Builder(memrefType).setShape(newShapeConstants);
345  assert(static_cast<int64_t>(newOperands.size()) ==
346  newMemRefType.getNumDynamicDims());
347 
348  // Create and insert the alloc op for the new memref.
349  auto newAlloc = rewriter.create<AllocLikeOp>(alloc.getLoc(), newMemRefType,
350  newOperands, IntegerAttr());
351  // Insert a cast so we have the same type as the old alloc.
352  auto resultCast = rewriter.create<MemRefCastOp>(alloc.getLoc(), newAlloc,
353  alloc.getType());
354 
355  rewriter.replaceOp(alloc, {resultCast});
356  return success();
357  }
358 };

564 /// Given a successor, try to collapse it to a new destination if it only
565 /// contains a passthrough unconditional branch. If the successor is
566 /// collapsable, `successor` and `successorOperands` are updated to reference
567 /// the new destination and values. `argStorage` is an optional storage to use
568 /// if operands to the collapsed successor need to be remapped.
569 static LogicalResult collapseBranch(Block *&successor,
570  ValueRange &successorOperands,
571  SmallVectorImpl<Value> &argStorage) {
572  // Check that the successor only contains a unconditional branch.
573  if (std::next(successor->begin()) != successor->end())
574  return failure();
575  // Check that the terminator is an unconditional branch.
576  BranchOp successorBranch = dyn_cast<BranchOp>(successor->getTerminator());
577  if (!successorBranch)
578  return failure();
579  // Check that the arguments are only used within the terminator.
580  for (BlockArgument arg : successor->getArguments()) {
581  for (Operation *user : arg.getUsers())
582  if (user != successorBranch)
583  return failure();
584  }
585  // Don't try to collapse branches to infinite loops.
586  Block *successorDest = successorBranch.getDest();
587  if (successorDest == successor)
588  return failure();
589 
590  // Update the operands to the successor. If the branch parent has no
591  // arguments, we can use the branch operands directly.
592  OperandRange operands = successorBranch.getOperands();
593  if (successor->args_empty()) {
594  successor = successorDest;
595  successorOperands = operands;
596  return success();
597  }
598 
599  // Otherwise, we need to remap any argument operands.
600  for (Value operand : operands) {
601  BlockArgument argOperand = operand.dyn_cast<BlockArgument>();
602  if (argOperand && argOperand.getOwner() == successor)
603  argStorage.push_back(successorOperands[argOperand.getArgNumber()]);
604  else
605  argStorage.push_back(operand);
606  }
607  successor = successorDest;
608  successorOperands = argStorage;
609  return success();
610 }

856 /// Compute `lhs` `pred` `rhs`, where `pred` is one of the known floating point
857 /// comparison predicates.
858 bool mlir::applyCmpPredicate(CmpFPredicate predicate, const APFloat &lhs,
859  const APFloat &rhs) {
860  auto cmpResult = lhs.compare(rhs);
861  switch (predicate) {
862  case CmpFPredicate::AlwaysFalse:
863  return false;
864  case CmpFPredicate::OEQ:
865  return cmpResult == APFloat::cmpEqual;
866  case CmpFPredicate::OGT:
867  return cmpResult == APFloat::cmpGreaterThan;
868  case CmpFPredicate::OGE:
869  return cmpResult == APFloat::cmpGreaterThan ||
870  cmpResult == APFloat::cmpEqual;
871  case CmpFPredicate::OLT:
872  return cmpResult == APFloat::cmpLessThan;
873  case CmpFPredicate::OLE:
874  return cmpResult == APFloat::cmpLessThan || cmpResult == APFloat::cmpEqual;
875  case CmpFPredicate::ONE:
876  return cmpResult != APFloat::cmpUnordered && cmpResult != APFloat::cmpEqual;
877  case CmpFPredicate::ORD:
878  return cmpResult != APFloat::cmpUnordered;
879  case CmpFPredicate::UEQ:
880  return cmpResult == APFloat::cmpUnordered || cmpResult == APFloat::cmpEqual;
881  case CmpFPredicate::UGT:
882  return cmpResult == APFloat::cmpUnordered ||
883  cmpResult == APFloat::cmpGreaterThan;
884  case CmpFPredicate::UGE:
885  return cmpResult == APFloat::cmpUnordered ||
886  cmpResult == APFloat::cmpGreaterThan ||
887  cmpResult == APFloat::cmpEqual;
888  case CmpFPredicate::ULT:
889  return cmpResult == APFloat::cmpUnordered ||
890  cmpResult == APFloat::cmpLessThan;
891  case CmpFPredicate::ULE:
892  return cmpResult == APFloat::cmpUnordered ||
893  cmpResult == APFloat::cmpLessThan || cmpResult == APFloat::cmpEqual;
894  case CmpFPredicate::UNE:
895  return cmpResult != APFloat::cmpEqual;
896  case CmpFPredicate::UNO:
897  return cmpResult == APFloat::cmpUnordered;
898  case CmpFPredicate::AlwaysTrue:
899  return true;
900  }
901  llvm_unreachable("unknown comparison predicate");
902 }

984 /// cond_br %cond, ^bb1(A, ..., N), ^bb1(A, ..., N)
985 /// -> br ^bb1(A, ..., N)
986 ///
987 /// cond_br %cond, ^bb1(A), ^bb1(B)
988 /// -> %select = select %cond, A, B
989 /// br ^bb1(%select)
990 ///
991 struct SimplifyCondBranchIdenticalSuccessors
992  : public OpRewritePattern<CondBranchOp> {
993  using OpRewritePattern<CondBranchOp>::OpRewritePattern;
994 
995  LogicalResult matchAndRewrite(CondBranchOp condbr,
996  PatternRewriter &rewriter) const override {
997  // Check that the true and false destinations are the same and have the same
998  // operands.
999  Block *trueDest = condbr.trueDest();
1000  if (trueDest != condbr.falseDest())
1001  return failure();
1002 
1003  // If all of the operands match, no selects need to be generated.
1004  OperandRange trueOperands = condbr.getTrueOperands();
1005  OperandRange falseOperands = condbr.getFalseOperands();
1006  if (trueOperands == falseOperands) {
1007  rewriter.replaceOpWithNewOp<BranchOp>(condbr, trueDest, trueOperands);
1008  return success();
1009  }
1010 
1011  // Otherwise, if the current block is the only predecessor insert selects
1012  // for any mismatched branch operands.
1013  if (trueDest->getUniquePredecessor() != condbr->getBlock())
1014  return failure();
1015 
1016  // Generate a select for any operands that differ between the two.
1017  SmallVector<Value, 8> mergedOperands;
1018  mergedOperands.reserve(trueOperands.size());
1019  Value condition = condbr.getCondition();
1020  for (auto it : llvm::zip(trueOperands, falseOperands)) {
1021  if (std::get<0>(it) == std::get<1>(it))
1022  mergedOperands.push_back(std::get<0>(it));
1023  else
1024  mergedOperands.push_back(rewriter.create<SelectOp>(
1025  condbr.getLoc(), condition, std::get<0>(it), std::get<1>(it)));
1026  }
1027 
1028  rewriter.replaceOpWithNewOp<BranchOp>(condbr, trueDest, mergedOperands);
1029  return success();
1030  }
1031 };

1140

1142  if (!value.getType().isa<NoneType>() && type != value.getType())
1143  return op.emitOpError() << "requires attribute's type (" << value.getType()
1144  << ") to match op's return type (" << type << ")";

1147  return success();

1160  "range of the integer result type");
1161  return success();

1165  if (!value.isa<FloatAttr>())
1166  return op.emitOpError("requires 'value' to be a floating point constant");
1167  return success();
1168  }
1169

1171  if (!value.isa<ElementsAttr>())
1172  return op.emitOpError("requires 'value' to be a shaped constant");
1173  return success();
1174  }

1380 OpFoldResult DimOp::fold(ArrayRef<Attribute> operands) {
1381  auto index = operands[1].dyn_cast_or_null<IntegerAttr>();
1382 
1383  // All forms of folding require a known index.
1384  if (!index)
1385  return {};
1386 
1387  auto argTy = memrefOrTensor().getType();
1388  // Fold if the shape extent along the given index is known.
1389  if (auto shapedTy = argTy.dyn_cast<ShapedType>()) {
1390  // Folding for unranked types (UnrankedMemRefType, UnrankedTensorType) is
1391  // not supported.
1392  if (!shapedTy.hasRank())
1393  return {};
1394  if (!shapedTy.isDynamicDim(index.getInt())) {
1395  Builder builder(getContext());
1396  return builder.getIndexAttr(shapedTy.getShape()[index.getInt()]);
1397  }
1398  }
1399 
1400  Operation *definingOp = memrefOrTensor().getDefiningOp();
1401  // dim(tensor_load(memref)) -> dim(memref)
1402  if (auto tensorLoadOp = dyn_cast_or_null<TensorLoadOp>(definingOp)) {
1403  setOperand(0, tensorLoadOp.memref());
1404  return getResult();
1405  }
1406 
1407  // Fold dim to the operand of dynamic_tensor_from_elements.
1408  if (auto fromElements =
1409  dyn_cast_or_null<DynamicTensorFromElementsOp>(definingOp)) {
1410  auto resultType =
1411  fromElements.getResult().getType().cast<RankedTensorType>();
1412  // The case where the type encodes the size of the dimension is handled
1413  // above.
1414  assert(resultType.getShape()[index.getInt()] ==
1415  RankedTensorType::kDynamicSize);
1416 
1417  // Find the operand of the fromElements that corresponds to this index.
1418  auto dynExtents = fromElements.dynamicExtents().begin();
1419  for (auto dim : resultType.getShape().take_front(index.getInt()))
1420  if (dim == RankedTensorType::kDynamicSize)
1421  dynExtents++;
1422 
1423  return Value{*dynExtents};
1424  }
1425 
1426  // Fold dim to the size argument for an `AllocOp`, `ViewOp`, or `SubViewOp`.
1427  auto memrefType = argTy.dyn_cast<MemRefType>();
1428  if (!memrefType)
1429  return {};
1430 
1431  // The size at the given index is now known to be a dynamic size of a memref.
1432  unsigned unsignedIndex = index.getValue().getZExtValue();
1433  if (auto alloc = dyn_cast_or_null<AllocOp>(definingOp))
1434  return *(alloc.getDynamicSizes().begin() +
1435  memrefType.getDynamicDimIndex(unsignedIndex));
1436 
1437  if (auto view = dyn_cast_or_null<ViewOp>(definingOp))
1438  return *(view.getDynamicSizes().begin() +
1439  memrefType.getDynamicDimIndex(unsignedIndex));
1440 
1441  if (auto subview = dyn_cast_or_null<SubViewOp>(definingOp)) {
1442  assert(subview.isDynamicSize(unsignedIndex) &&
1443  "Expected dynamic subview size");
1444  return subview.getDynamicSize(unsignedIndex);
1445  }
1446 
1447  // dim(memrefcast) -> dim
1448  if (succeeded(foldMemRefCast(*this)))
1449  return getResult();
1450 
1451  return {};
1452 }

1521 ParseResult DmaStartOp::parse(OpAsmParser &parser, OperationState &result) {
1522  OpAsmParser::OperandType srcMemRefInfo;
1523  SmallVector<OpAsmParser::OperandType, 4> srcIndexInfos;
1524  OpAsmParser::OperandType dstMemRefInfo;
1525  SmallVector<OpAsmParser::OperandType, 4> dstIndexInfos;
1526  OpAsmParser::OperandType numElementsInfo;
1527  OpAsmParser::OperandType tagMemrefInfo;
1528  SmallVector<OpAsmParser::OperandType, 4> tagIndexInfos;
1529  SmallVector<OpAsmParser::OperandType, 2> strideInfo;
1530 
1531  SmallVector<Type, 3> types;
1532  auto indexType = parser.getBuilder().getIndexType();
1533 
1534  // Parse and resolve the following list of operands:
1535  // *) source memref followed by its indices (in square brackets).
1536  // *) destination memref followed by its indices (in square brackets).
1537  // *) dma size in KiB.
1538  if (parser.parseOperand(srcMemRefInfo) ||
1539  parser.parseOperandList(srcIndexInfos, OpAsmParser::Delimiter::Square) ||
1540  parser.parseComma() || parser.parseOperand(dstMemRefInfo) ||
1541  parser.parseOperandList(dstIndexInfos, OpAsmParser::Delimiter::Square) ||
1542  parser.parseComma() || parser.parseOperand(numElementsInfo) ||
1543  parser.parseComma() || parser.parseOperand(tagMemrefInfo) ||
1544  parser.parseOperandList(tagIndexInfos, OpAsmParser::Delimiter::Square))
1545  return failure();
1546 
1547  // Parse optional stride and elements per stride.
1548  if (parser.parseTrailingOperandList(strideInfo))
1549  return failure();
1550 
1551  bool isStrided = strideInfo.size() == 2;
1552  if (!strideInfo.empty() && !isStrided) {
1553  return parser.emitError(parser.getNameLoc(),
1554  "expected two stride related operands");
1555  }
1556 
1557  if (parser.parseColonTypeList(types))
1558  return failure();
1559  if (types.size() != 3)
1560  return parser.emitError(parser.getNameLoc(), "fewer/more types expected");
1561 
1562  if (parser.resolveOperand(srcMemRefInfo, types[0], result.operands) ||
1563  parser.resolveOperands(srcIndexInfos, indexType, result.operands) ||
1564  parser.resolveOperand(dstMemRefInfo, types[1], result.operands) ||
1565  parser.resolveOperands(dstIndexInfos, indexType, result.operands) ||

1567  parser.resolveOperand(numElementsInfo, indexType, result.operands) ||
1568  parser.resolveOperand(tagMemrefInfo, types[2], result.operands) ||
1569  // tag indices should be index.
1570  parser.resolveOperands(tagIndexInfos, indexType, result.operands))
1571  return failure();
1572 
1573  if (isStrided) {
1574  if (parser.resolveOperands(strideInfo, indexType, result.operands))
1575  return failure();
1576  }
1577 
1578  return success();
1579 }

1598  !llvm::all_of(getSrcIndices().getTypes(),

1600  return emitOpError("expected source indices to be of index type");
1601 
1602  // 2. Destination memref.

1604  return emitOpError("expected destination to be of memref type");
1605  unsigned numExpectedOperands = getSrcMemRefRank() + getDstMemRefRank() + 4;
1606  if (numOperands < numExpectedOperands)
1607  return emitOpError() << "expected at least " << numExpectedOperands
1608  << " operands";
1609  if (!getDstIndices().empty() &&
1610  !llvm::all_of(getDstIndices().getTypes(),
1611  [](Type t) { return t.isIndex(); }))
1612  return emitOpError("expected destination indices to be of index type");

1616  return emitOpError("expected num elements to be of index type");
1617 
1618  // 4. Tag memref.
1619  if (!getTagMemRef().getType().isa<MemRefType>())
1620  return emitOpError("expected tag to be of memref type");
1621  numExpectedOperands += getTagMemRefRank();
1622  if (numOperands < numExpectedOperands)
1623  return emitOpError() << "expected at least " << numExpectedOperands

1631  if (getSrcMemorySpace() == getDstMemorySpace())
1632  return emitOpError("DMA should be between different memory spaces");
