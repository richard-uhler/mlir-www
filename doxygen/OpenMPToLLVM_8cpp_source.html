19 /// A pattern that converts the region arguments in a single-region OpenMP
20 /// operation to the LLVM dialect. The body of the region is not modified and is
21 /// expected to either be processed by the conversion infrastructure or already
22 /// contain ops compatible with LLVM dialect types.
23 template <typename OpType>
24 struct RegionOpConversion : public ConvertOpToLLVMPattern<OpType> {
25  using ConvertOpToLLVMPattern<OpType>::ConvertOpToLLVMPattern;
26 
27  LogicalResult
28  matchAndRewrite(OpType curOp, ArrayRef<Value> operands,
29  ConversionPatternRewriter &rewriter) const override {
30  auto newOp = rewriter.create<OpType>(curOp.getLoc(), TypeRange(), operands,
31  curOp.getAttrs());
32  rewriter.inlineRegionBefore(curOp.region(), newOp.region(),
33  newOp.region().end());
34  if (failed(rewriter.convertRegionTypes(&newOp.region(),
35  *this->getTypeConverter())))
36  return failure();
37 
38  rewriter.eraseOp(curOp);
39  return success();
40  }
41 };
42 } // namespace

57 void ConvertOpenMPToLLVMPass::runOnOperation() {
58  auto module = getOperation();
59 
60  // Convert to OpenMP operations with LLVM IR dialect
61  OwningRewritePatternList patterns;
62  LLVMTypeConverter converter(&getContext());
63  populateStdToLLVMConversionPatterns(converter, patterns);
64  populateOpenMPToLLVMConversionPatterns(converter, patterns);
65 
66  LLVMConversionTarget target(getContext());
67  target.addDynamicallyLegalOp<omp::ParallelOp, omp::WsLoopOp>(
68  [&](Operation *op) { return converter.isLegal(&op->getRegion(0)); });
69  target.addLegalOp<omp::TerminatorOp, omp::TaskyieldOp, omp::FlushOp,
70  omp::BarrierOp, omp::TaskwaitOp>();
71  if (failed(applyPartialConversion(module, target, std::move(patterns))))
72  signalPassFailure();
73 }
