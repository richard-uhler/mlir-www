112 llvm::Optional<StringRef> TypeParameter::getAllocator() const {
113  llvm::Init *parameterType = def->getArg(num);
114  if (isa<llvm::StringInit>(parameterType))
115  return llvm::Optional<StringRef>();
116 
117  if (auto *typeParameter = dyn_cast<llvm::DefInit>(parameterType)) {
118  llvm::RecordVal *code = typeParameter->getDef()->getValue("allocator");
119  if (!code)
120  return llvm::Optional<StringRef>();
121  if (llvm::StringInit *ci = dyn_cast<llvm::StringInit>(code->getValue()))
122  return ci->getValue();
123  if (isa<llvm::UnsetInit>(code->getValue()))
124  return llvm::Optional<StringRef>();
125 
126  llvm::PrintFatalError(
