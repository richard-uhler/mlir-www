28 // Create a new call to the type-canonicalized `LinalgOp::getLibraryCallName()`
29 // function. The implementation of the function can be either in the same module
30 // or in an externally linked library.
31 // This is a generic entry point for all LinalgOp, except for CopyOp and
32 // IndexedGenericOp, for which omre specialized patterns are provided.
33 class LinalgOpToLibraryCallRewrite : public RewritePattern {
34 public:
35  LinalgOpToLibraryCallRewrite()

37 
38  LogicalResult matchAndRewrite(Operation *op,
39  PatternRewriter &rewriter) const override;
40 };
