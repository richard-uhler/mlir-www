37 // A helper RAII class to emit nested namespaces for this op.
38 class NamespaceEmitter {
39 public:
40  NamespaceEmitter(raw_ostream &os, const Dialect &dialect) : os(os) {
41  if (!dialect)
42  return;
43  llvm::SplitString(dialect.getCppNamespace(), namespaces, "::");
44  for (StringRef ns : namespaces)
45  os << "namespace " << ns << " {\n";
46  }
47 
48  ~NamespaceEmitter() {
49  for (StringRef ns : llvm::reverse(namespaces))
50  os << "} // namespace " << ns << "\n";
51  }
52 
53 private:
54  raw_ostream &os;
55  SmallVector<StringRef, 2> namespaces;
56 };
